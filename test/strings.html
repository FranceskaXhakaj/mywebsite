def checkTwos(s):

    return s[1] == s[len(s) - 2]

[10.21] Given a string of numbers and characters, write the function numsOnly that returns all the numbers in the string in the same order they occur (in string form).


def numsOnly(s):
    result = ""
    for char in s:
        if char.isdigit():
            result += char
    return result

print(numsOnly("o23yay9"))  # Output: "239"
print(numsOnly("hello"))     # Output: ""
print(numsOnly("I have 1 infinite loop, 5 assertion errors, 11 bugs and 0 motivation right now."))  # Output: "15110"

[10.22] Write a function matchString that takes in two strings, one a sentence and a word, and returns how many times the word occurs in the sentence. Be careful of casing - we are counting the word regardless of if it is lowercase vs uppercase. 


def matchString(sentence, word):
    sentenceLower = sentence.lower()
    wordLower = word.lower()
    return sentenceLower.count(wordLower)

print(matchString("Programming is fun, programming is creative", "programming")) # Output: 2 
print(matchString("The cat with the red hat on the blue mat.", "the")) # Output: 3

[10.23] Write the function numFun that takes a string as a paramter. The function should print the number of digits in the string and should return the average of these numbers. 


def numFun(s):
    total = 0
    count = 0
    
    for i in range(len(s)):
        if s[i].isdigit():
            total += int(s[i])
            count += 1
    
    avg = total / count
    
    print("Count:", count)
    
    return avg

print(numFun("PYnative29@#8496"))   # Prints 6, Returns: 6.333333333333333
print(numFun("jhfbjadj198ksjn98")) # Prints 5, Returns: 7.0

[10.26] Write the function not_string that takes a string as a paramter and returns a new string where the string "not " has been added to the front. However, if the string already begins with "not", return the string unchanged. 


def not_string(s):
    if s.startswith("not"):
        return s
    else:
        return "not " + s

# Test cases
print(not_string('candy'))    # Output: 'not candy'
print(not_string('x'))        # Output: 'not x'
print(not_string('not bad'))  # Output: 'not bad'

[10.27] Write the function count_characters that takes a string as a parameter and counts and prints all letters, digits, and special symbols from the given string. 


def count_characters(s):
    chars = 0
    digits = 0
    symbols = 0

    for i in range(len(s)):
        if s[i] in "abcdefghijklmnopqrstuvwxyz" or s[i] in "ABCDEFGHIJKLMNOPQRSTUVWXYZ":
            chars += 1
        elif s[i].isdigit():
            digits += 1
        else:
            symbols += 1

    print("Chars =", chars)
    print("Digits =", digits)
    print("Symbols =", symbols)

# Test the function
str1 = "P@#yn26at^&i5ve"
count_characters(str1)

[10.28] Write a function count_vowels that takes a sentence as paramter and counts the number of vowels in the sentence (a, e, i, o, u). Count for both upper and lower case vowels. 


def count_vowels(sentence):
    vowels = "aeiou"
    count = 0
    
    # Convert the sentence to lowercase to handle case-insensitivity
    sentence = sentence.lower()
    
    # Iterate through each character in the sentence
    for i in range(len(sentence)):
        # Check if the character is a vowel
        if sentence[i] in vowels:
            count += 1
    
    return count

# Test the function
sentence = input("Enter a sentence: ")
num_vowels = count_vowels(sentence)
print("Number of vowels in the sentence:", num_vowels)
